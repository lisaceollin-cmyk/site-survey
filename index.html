function doGet(e) { return ContentService .createTextOutput('場勘上傳服務運作中 ✓') .setMimeType(ContentService.MimeType.TEXT); } // ===== 場勘紀錄 Drive 上傳 webhook ===== // 總資料夾名稱（要改存放位置就改這行） var ROOT_FOLDER = '場勘紀錄'; function doPost(e) { try { var data = JSON.parse(e.postData.contents); var vendor = (data.vendor || '未命名廠商').trim(); var zone = (data.zone || '未命名區域').trim(); var notes = data.notes || ''; var filename = data.filename || ('file_' + Date.now()); var mimeType = data.mimeType || 'image/jpeg'; var bytes = Utilities.base64Decode(data.data); // 逐層找/建資料夾：場勘紀錄 → 廠商 → 區域 var root = getOrCreateFolder(DriveApp.getRootFolder(), ROOT_FOLDER); var vFold = getOrCreateFolder(root, vendor); var zFold = getOrCreateFolder(vFold, zone); var blob = Utilities.newBlob(bytes, mimeType, filename); // 同名檢查：若資料夾已有同名檔，先丟到垃圾桶（同一份重傳不會增生） var dup = zFold.getFilesByName(filename); var replaced = false; while (dup.hasNext()) { dup.next().setTrashed(true); replaced = true; } var file = zFold.createFile(blob); // 有備註就寫一份 txt（同名覆蓋） if (notes) { var existing = zFold.getFilesByName('備註.txt'); while (existing.hasNext()) { existing.next().setTrashed(true); } zFold.createFile('備註.txt', notes, 'text/plain'); } return json({ ok: true, fileId: file.getId(), name: filename, replaced: replaced }); } catch (err) { return json({ ok: false, error: String(err) }); } } // 在 parent 底下找名為 name 的資料夾，沒有就建一個 function getOrCreateFolder(parent, name) { var it = parent.getFoldersByName(name); return it.hasNext() ? it.next() : parent.createFolder(name); } function json(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
